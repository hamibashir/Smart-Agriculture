// Device Configuration

// ============================================
// Pin Definitions
// ============================================
#define SOIL_PIN  34
#define RAIN_PIN  33
#define LIGHT_PIN 32
#define DHT_PIN   4
#define RELAY_PIN 5

#define DHT_TYPE DHT22
DHT dht(DHT_PIN, DHT_TYPE);

// ============================================
// Sensor Thresholds
// ============================================
#define SOIL_DRY_THRESHOLD   2800  // ADC > this = dry  (~32% moisture)
#define SOIL_WET_THRESHOLD   1200  // ADC < this = wet  (~70% moisture)
#define RAIN_THRESHOLD       2000  // ADC < this = raining
#define LIGHT_DARK_THRESHOLD 3000

// ============================================
// Timing Configuration — 3-Tier System
// ============================================
//
//  Tier 1 — Sensor read + local rain safety    (every 1 second)
//    → Reads all sensors
//    → If rain detected: instantly turns pump OFF
//    → Fast, local, zero network calls
//
//  Tier 2 — Command poll from backend          (every 5 seconds)
//    → Lightweight GET /api/sensors/command/:id
//    → Syncs relay to whatever app commanded
//    → Zero DB writes on backend, very fast
//
//  Tier 3 — Full sensor data upload            (every 30 seconds)
//    → Full POST /api/sensors/reading
//    → Stores reading, triggers alerts & AI
//    → Also syncs pump state as fallback
//
#define SENSOR_READ_INTERVAL   500    // 0.5s — extremely fast, local safety
#define COMMAND_POLL_INTERVAL  1000   // 1s  — instant app response
#define SENSOR_UPLOAD_INTERVAL 2000   // 2s  — fast full data upload for FYP demo
#define BACKEND_RETRY_INTERVAL 2000   // 2s  — retry after failed upload
#define WIFI_RETRY_INTERVAL_MS 5000   // 5s  — retry WiFi if disconnected
#define PUMP_MIN_RUN_TIME      1000   // 1s  — min run time (offline)

#define ANALOG_SAMPLES 4

// ============================================
// State Variables
// ============================================
unsigned long lastSensorReadTime  = 0;
unsigned long lastCommandPollTime = 0;
unsigned long lastUploadTime      = 0;
unsigned long lastUploadAttempt   = 0;
unsigned long lastWifiAttemptMs   = 0;
unsigned long pumpStartTime       = 0;

bool pumpRunning   = false;
bool wifiConnected = false;

// Last pump state received from backend (-2 = never received)
int lastBackendPumpState = -2;

// Offline restore state
bool requestedPumpState = false;
bool localSafetyOverrideActive = false;

// Cached sensor values (shared between tiers)
int  cachedSoilValue  = 0;
int  cachedRainValue  = 4095;
int  cachedLightValue = 0;
float cachedTemp      = -999;
float cachedHum       = -999;

// ============================================
// Forward Declarations
// ============================================
static int   readAnalogSmoothed(uint8_t pin);
static bool  readDht(float& tempC, float& humPct);
static void  readAllSensors(int& soil, int& rain, int& light, float& tempC, float& humPct);
static void  displaySensorData(int soil, int rain, int light, float temp, float hum);
static void  applyLocalSafety(int rainValue, int soilValue);
static void  runOfflinePumpLogic(int soilValue, int rainValue);
static void  syncRelayToCommand(int commanded, const char* reason);
static void  turnPumpOn();
static void  turnPumpOff();
static void  connectToWiFi();
static float convertSoilMoistureToPercentage(int rawValue);
static float convertLightToPercentage(int rawValue);
static bool  fetchPumpCommand();
static bool  sendSensorDataToBackend(int soilRaw, int rainRaw, int lightRaw, float temp, float hum);
static void  printWifiDiag();

// ============================================
// Setup
// ============================================
void setup() {
  Serial.begin(115200);
  delay(2000);

  Serial.println("\n====================================");
  Serial.println(" ESP32 Smart Irrigation v2 (Fast)");
  Serial.println("====================================\n");

  dht.begin();
  Serial.println("DHT initialized");

  // Active-LOW relay: HIGH = OFF (safe default)
  pinMode(RELAY_PIN, OUTPUT_OPEN_DRAIN);
  digitalWrite(RELAY_PIN, HIGH);
  Serial.println("Relay OFF (safe default)");

  Serial.println("Analog inputs ready");
  Serial.println("\nConnecting to WiFi...");
  connectToWiFi();

  if (wifiConnected) {
    Serial.println("WiFi OK");
    printWifiDiag();
    Serial.printf("RSSI: %d dBm\n", WiFi.RSSI());
    Serial.printf("device_id: %s\n", deviceId);
    Serial.printf("Upload  : %s%s%s\n", BACKEND_USE_HTTPS ? "https://" : "http://", BACKEND_HOST, API_PATH);
    Serial.printf("Commands: %s%s%s\n", BACKEND_USE_HTTPS ? "https://" : "http://", BACKEND_HOST, COMMAND_PATH);
  } else {
    Serial.println("WiFi failed - offline sensor logic will run");
  }

  Serial.println("\n--- 3-Tier Timing ---");
  Serial.println("Tier 1: Sensors + Rain safety  (1 second)");
  Serial.println("Tier 2: Command poll           (5 seconds)");
  Serial.println("Tier 3: Full sensor upload     (30 seconds)");
  Serial.println("\nStarting loop.\n");
  delay(1000);
}

// ============================================
// Main Loop — 3-Tier System
// ============================================
void loop() {
  unsigned long now = millis();

  // --- WiFi health check ---
  if (WiFi.status() == WL_CONNECTED) {
    wifiConnected = true;
    lastWifiAttemptMs = now;
  } else {
    wifiConnected = false;
    if (now - lastWifiAttemptMs >= WIFI_RETRY_INTERVAL_MS) {
      Serial.println("WiFi down - reconnecting...");
      connectToWiFi();
      lastWifiAttemptMs = millis();
    }
  }

  // ── TIER 1: Sensor read + local safety (every 1 second) ──────────────────
  // This runs regardless of WiFi. Rain is handled INSTANTLY here.
  bool doRead = (now - lastSensorReadTime >= SENSOR_READ_INTERVAL);
  if (doRead) {
    lastSensorReadTime = now;
    readAllSensors(cachedSoilValue, cachedRainValue, cachedLightValue, cachedTemp, cachedHum);
    displaySensorData(cachedSoilValue, cachedRainValue, cachedLightValue, cachedTemp, cachedHum);

    if (wifiConnected) {
      // Online: backend controls the pump, BUT safety is applied locally
      // so the relay turns off INSTANTLY without waiting for the poll.
      applyLocalSafety(cachedRainValue, cachedSoilValue);
    } else {
      // Offline: full sensor-based decision logic runs here
      runOfflinePumpLogic(cachedSoilValue, cachedRainValue);
    }
  }

  // ── TIER 2: Command poll from backend (every 5 seconds) ──────────────────
  // Lightweight GET — zero DB writes. Syncs relay instantly to app commands.
  bool doCommandPoll = wifiConnected && (now - lastCommandPollTime >= COMMAND_POLL_INTERVAL);
  if (doCommandPoll) {
    lastCommandPollTime = now;
    fetchPumpCommand();
  }

  // ── TIER 3: Full sensor data upload (every 30 seconds) ───────────────────
  // Stores readings in DB, triggers AI & alerts. Also syncs pump as fallback.
  bool enoughSinceSuccess = (now - lastUploadTime    >= SENSOR_UPLOAD_INTERVAL);
  bool enoughSinceAttempt = (now - lastUploadAttempt >= BACKEND_RETRY_INTERVAL);
  bool doUpload = wifiConnected && enoughSinceSuccess && enoughSinceAttempt;
  if (doUpload) {
    lastUploadAttempt = now;
    bool ok = sendSensorDataToBackend(cachedSoilValue, cachedRainValue, cachedLightValue, cachedTemp, cachedHum);
    if (ok) lastUploadTime = now;
    Serial.println(ok ? "Upload: OK\n" : "Upload: FAIL\n");
  }

  if (!doRead && !doCommandPoll && !doUpload) {
    delay(50);
  }
}

// ============================================
// Tier 1 Helper: Instant local safety (Rain + Soil)
// Turns relay OFF immediately when rain or wet soil is detected.
// Sets localSafetyOverrideActive flag so Tier 2 polls
// cannot turn the pump back ON while condition persists.
// ============================================
void applyLocalSafety(int rainValue, int soilValue) {
  bool raining = (rainValue < RAIN_THRESHOLD);
  bool soilWet = (soilValue < SOIL_WET_THRESHOLD);

  if (raining || soilWet) {
    if (!localSafetyOverrideActive) {
      // Condition just started — force OFF and lock the flag
      localSafetyOverrideActive = true;
      turnPumpOff();
      Serial.println("LOCAL SAFETY: Rain or Wet Soil detected — pump OFF instantly, override locked");
      // Keeping lastBackendPumpState intact ensures proper re-sync later
    }
    // Condition still active — do nothing
    return;
  }

  // ── Condition just cleared ───────────────────────────────────────────────────
  if (localSafetyOverrideActive) {
    localSafetyOverrideActive = false;
    Serial.println("LOCAL SAFETY: Condition cleared — restoring pump state");
    // Re-evaluate: force syncRelayToCommand to apply lastBackendPumpState again
    int restoreCmd = lastBackendPumpState;
    lastBackendPumpState = -2;
    syncRelayToCommand(restoreCmd, "safety_cleared_restore");
  }
}

// ============================================
// Tier 2: Lightweight command fetch from backend
// GET /api/sensors/command/:deviceId
// Zero DB writes. Returns pump_status + pump_reason.
// ============================================
bool fetchPumpCommand() {
  if (WiFi.status() != WL_CONNECTED) return false;

  WiFiClientSecure secureClient;
  WiFiClient       plainClient;
  HTTPClient       http;

  String endpoint;
  bool beginOk = false;

  if (BACKEND_USE_HTTPS) {
    endpoint = String("https://") + BACKEND_HOST + COMMAND_PATH + deviceId;
    secureClient.setInsecure();
    beginOk = http.begin(secureClient, endpoint);
  } else {
    endpoint = String("http://") + BACKEND_HOST + ":" + String(BACKEND_PORT) + COMMAND_PATH + deviceId;
    beginOk = http.begin(plainClient, endpoint);
  }

  if (!beginOk) return false;

  http.setConnectTimeout(1500); // Reduced from 3000ms to prevent loop blocking
  http.setTimeout(2000);        // Reduced from 4000ms

  int code = http.GET();
  bool success = false;

  if (code == 200) {
    String response = http.getString();
    StaticJsonDocument<256> doc; // Increased from 128 to prevent NoMemory errors
    if (!deserializeJson(doc, response) && doc.containsKey("pump_status")) {
      int commanded       = doc["pump_status"].as<int>();
      const char* reason  = doc["pump_reason"] | "unknown";
      syncRelayToCommand(commanded, reason);
      success = true;
    }
  } else {
    Serial.printf("CMD poll failed: HTTP %d\n", code);
  }

  http.end();
  return success;
}

// ============================================
// Relay Sync — Only acts when state changes
// Prevents repeated toggling from repeated polls.
// rainOverrideActive gate: blocks any 'on' command
// from the backend while rain is locally detected.
// ============================================
void syncRelayToCommand(int commanded, const char* reason) {
  // ── Safety gate ───────────────────────────────────────────────────────────
  // If safety override is active locally, block backend 'on' commands.
  if (localSafetyOverrideActive && commanded == 1) {
    Serial.printf("SYNC BLOCKED: backend says ON but local safety override active [%s]\n", reason);
    return;  // Ignore — relay stays OFF, lastBackendPumpState unchanged
  }

  if (commanded == lastBackendPumpState) {
    return;  // No change — do nothing
  }

  lastBackendPumpState = commanded;

  if (commanded == 1 && !pumpRunning) {
    turnPumpOn();
    Serial.printf("SYNC ON  <- backend [%s]\n", reason);
  } else if (commanded == 0 && pumpRunning) {
    turnPumpOff();
    Serial.printf("SYNC OFF <- backend [%s]\n", reason);
  }
}

// ============================================
// Sensor Reading
// ============================================
int readAnalogSmoothed(uint8_t pin) {
  long sum = 0;
  for (int i = 0; i < ANALOG_SAMPLES; i++) {
    sum += analogRead(pin);
    delayMicroseconds(150);
  }
  return (int)(sum / ANALOG_SAMPLES);
}

bool readDht(float& tempC, float& humPct) {
  humPct = dht.readHumidity();
  tempC  = dht.readTemperature();
  if (isnan(humPct) || isnan(tempC)) {
    tempC  = -999;
    humPct = -999;
    return false;
  }
  return true;
}

void readAllSensors(int& soil, int& rain, int& light, float& tempC, float& humPct) {
  soil  = readAnalogSmoothed(SOIL_PIN);
  rain  = readAnalogSmoothed(RAIN_PIN);
  light = readAnalogSmoothed(LIGHT_PIN);
  readDht(tempC, humPct);
}

// ============================================
// Display
// ============================================
void displaySensorData(int soil, int rain, int light, float temp, float hum) {
  Serial.println("-------------------------------------");
  Serial.print("Soil ADC: "); Serial.print(soil);
  if (soil > SOIL_DRY_THRESHOLD)   Serial.println("  (dry)");
  else if (soil > 1500)             Serial.println("  (moderate)");
  else                              Serial.println("  (wet)");

  Serial.print("Rain ADC: "); Serial.print(rain);
  Serial.println(rain < RAIN_THRESHOLD ? "  (RAINING)" : "  (no rain)");

  Serial.print("Light ADC: "); Serial.print(light);
  if (light > LIGHT_DARK_THRESHOLD) Serial.println("  (dark)");
  else if (light > 1500)            Serial.println("  (moderate)");
  else                              Serial.println("  (bright)");

  Serial.print("Temp: ");
  if (temp != -999) { Serial.print(temp, 1); Serial.println(" C"); }
  else Serial.println("ERROR");

  Serial.print("Humidity: ");
  if (hum != -999) { Serial.print(hum, 1); Serial.println(" %"); }
  else Serial.println("ERROR");

  Serial.print("Pump: ");
  Serial.print(pumpRunning ? "ON" : "OFF");
  Serial.print("  [");
  Serial.print(wifiConnected ? "ONLINE — backend+local rain" : "OFFLINE — sensor logic");
  Serial.println("]");
  Serial.println("-------------------------------------");
}

// ============================================
// Offline Fallback Pump Logic
// Full P1-P5 priority chain, same as backend.
// Runs only when WiFi is disconnected.
// ============================================
void runOfflinePumpLogic(int soilValue, int rainValue) {
  bool raining = (rainValue < RAIN_THRESHOLD);
  bool soilWet = (soilValue < SOIL_WET_THRESHOLD);
  bool soilDry = (soilValue > SOIL_DRY_THRESHOLD);

  // Update requested state based on current sensors
  bool shouldPumpRun = requestedPumpState;
  if (soilDry)      shouldPumpRun = true;
  else if (soilWet) shouldPumpRun = false;
  requestedPumpState = shouldPumpRun;

  // P1 — Rain override (temporary)
  if (raining) {
    if (!localSafetyOverrideActive) {
      localSafetyOverrideActive = true;
      Serial.println("OFFLINE P1: Rain detected — pump forced OFF (intent preserved)");
    }
    if (pumpRunning) turnPumpOff();
    return;
  }

  // Rain just cleared — re-evaluate
  if (localSafetyOverrideActive) {
    localSafetyOverrideActive = false;
    Serial.println("OFFLINE: Rain cleared — restoring pump state...");
  }

  // P2 — Soil wet override (temporary)
  if (soilWet) {
    if (pumpRunning && millis() - pumpStartTime >= PUMP_MIN_RUN_TIME) {
      turnPumpOff();
      Serial.println("OFFLINE P2: Pump OFF (soil moisture sufficient)");
    }
    return;
  }

  // P4/Restore — Apply requested state
  if (requestedPumpState && !pumpRunning) {
    turnPumpOn();
    Serial.println("OFFLINE P4: Pump ON (dry soil / restoring after rain)");
  } else if (!requestedPumpState && pumpRunning) {
    if (millis() - pumpStartTime >= PUMP_MIN_RUN_TIME) {
      turnPumpOff();
      Serial.println("OFFLINE P5: Pump OFF");
    }
  }
}

// ============================================
// Relay Control
// ============================================
void turnPumpOn() {
  digitalWrite(RELAY_PIN, LOW);   // Active-LOW relay
  pumpRunning   = true;
  pumpStartTime = millis();
}

void turnPumpOff() {
  digitalWrite(RELAY_PIN, HIGH);  // Active-LOW relay
  pumpRunning = false;
}

// ============================================
// WiFi
// ============================================
void connectToWiFi() {
  if (WiFi.status() == WL_CONNECTED) {
    wifiConnected = true;
    return;
  }
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  Serial.print("WiFi");
  unsigned long startTime = millis();
  while (WiFi.status() != WL_CONNECTED && (millis() - startTime < 12000)) {
    delay(400);
    Serial.print(".");
  }
  Serial.println();

  wifiConnected = (WiFi.status() == WL_CONNECTED);
  if (!wifiConnected) Serial.println("WiFi connect failed");
}

// ============================================
// Conversions
// ============================================
float convertSoilMoistureToPercentage(int rawValue) {
  return constrain((float)map(rawValue, 0, 4095, 100, 0), 0.0f, 100.0f);
}

float convertLightToPercentage(int rawValue) {
  return constrain((float)map(rawValue, 0, 4095, 100, 0), 0.0f, 100.0f);
}

// ============================================
// WiFi Diagnostics
// ============================================
void printWifiDiag() {
  if (WiFi.status() != WL_CONNECTED) return;
  Serial.printf("ESP32 IP: %s\n", WiFi.localIP().toString().c_str());
  Serial.printf("Gateway:  %s\n", WiFi.gatewayIP().toString().c_str());

  IPAddress resolved;
  if (WiFi.hostByName(BACKEND_HOST, resolved)) {
    Serial.printf("Backend:  %s\n", resolved.toString().c_str());
  } else {
    Serial.println("DNS lookup failed for backend host");
  }
}

// ============================================
// Tier 3: Full sensor data upload (POST)
// Stores reading in DB, triggers AI & alerts.
// Also syncs pump state as a fallback in case
// the 5s command poll missed something.
// ============================================
bool sendSensorDataToBackend(int soilRaw, int rainRaw, int lightRaw, float temp, float hum) {
  if (WiFi.status() != WL_CONNECTED) return false;

  WiFiClientSecure secureClient;
  WiFiClient       plainClient;
  HTTPClient       http;

  float soilMoisture   = convertSoilMoistureToPercentage(soilRaw);
  float lightIntensity = convertLightToPercentage(lightRaw);
  bool  raining        = (rainRaw < RAIN_THRESHOLD);

  StaticJsonDocument<384> doc;
  doc["device_id"]       = deviceId;
  doc["soil_moisture"]   = soilMoisture;
  doc["light_intensity"] = lightIntensity;
  if (temp != -999) doc["temperature"] = temp;
  if (hum  != -999) doc["humidity"]    = hum;
  doc["rainfall"] = raining;
  doc["pump_on"]  = pumpRunning ? 1 : 0;

  String jsonString;
  serializeJson(doc, jsonString);
  Serial.println("UPLOAD sensor reading...");
  Serial.println(jsonString);

  String endpoint;
  bool beginOk = false;
  if (BACKEND_USE_HTTPS) {
    endpoint = String("https://") + BACKEND_HOST + API_PATH;
    secureClient.setInsecure();
    beginOk = http.begin(secureClient, endpoint);
  } else {
    endpoint = String("http://") + BACKEND_HOST + ":" + String(BACKEND_PORT) + API_PATH;
    beginOk = http.begin(plainClient, endpoint);
  }

  if (!beginOk) {
    Serial.println("HTTP begin() failed");
    return false;
  }

  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);
  const char* headerKeys[] = {"Location"};
  http.collectHeaders(headerKeys, 1);
  http.addHeader("Content-Type", "application/json");
  http.setConnectTimeout(3000); // Reduced from 8000ms to keep loop fast
  http.setTimeout(4000);        // Reduced from 10000ms

  int  code    = http.POST(jsonString);
  bool success = false;

  if (code > 0) {
    Serial.printf("HTTP %d\n", code);
    if (code == 301 || code == 302 || code == 307 || code == 308) {
      Serial.print("Redirect to: ");
      Serial.println(http.header("Location"));
    }
    String response = http.getString();
    if (response.length()) {
      Serial.println(response);
      // Fallback sync — command poll handles this faster, but this catches anything missed
      StaticJsonDocument<512> respDoc; // Increased from 256 to handle full response without NoMemory error
      if (!deserializeJson(respDoc, response) && respDoc.containsKey("pump_status")) {
        int commanded      = respDoc["pump_status"].as<int>();
        const char* reason = respDoc["pump_reason"] | "upload_sync";
        syncRelayToCommand(commanded, reason);
      }
    }
    success = (code == 200 || code == 201);
  } else {
    Serial.printf("HTTP error %d: %s\n", code, http.errorToString(code).c_str());
  }

  http.end();
  return success;
}
