#include <WiFi.h>
#include <WiFiClientSecure.h>
#include <HTTPClient.h>
#include <ArduinoJson.h>
#include <DHT.h>

// ============================================
// WiFi Configuration
// ============================================
const char* ssid = "Hamza";
const char* password = "hamza2004";

// ============================================
// Backend API Configuration (Production)
// ============================================
const char* BACKEND_HOST = "api.hamzabashir.online";
const uint16_t BACKEND_PORT = 443;
const char* API_PATH = "/api/sensors/reading/";
const bool BACKEND_USE_HTTPS = true;

// ============================================
// Device Configuration
// ============================================
const char* deviceId = "ESP_32test";

// ============================================
// Pin Definitions
// ============================================
#define SOIL_PIN 34
#define RAIN_PIN 33
#define LIGHT_PIN 32
#define DHT_PIN 4
#define RELAY_PIN 14

#define DHT_TYPE DHT22
DHT dht(DHT_PIN, DHT_TYPE);

// ============================================
// Sensor Thresholds
// ============================================
#define SOIL_DRY_THRESHOLD 2500
#define RAIN_THRESHOLD 2000
#define LIGHT_DARK_THRESHOLD 3000

// ============================================
// Timing Configuration
// ============================================
#define READING_INTERVAL 3000
#define BACKEND_SEND_INTERVAL 30000
#define BACKEND_RETRY_INTERVAL 5000
#define PUMP_MIN_RUN_TIME 1500
#define WIFI_RETRY_INTERVAL_MS 10000

#define ANALOG_SAMPLES 4

unsigned long lastReadTime = 0;
unsigned long lastBackendSendTime = 0;
unsigned long lastBackendAttemptTime = 0;
unsigned long pumpStartTime = 0;
unsigned long lastWifiAttemptMs = 0;
bool pumpRunning = false;
bool wifiConnected = false;

static int readAnalogSmoothed(uint8_t pin);
static bool readDht(float& tempC, float& humPct);
static void readAllSensors(int& soil, int& rain, int& light, float& tempC, float& humPct);
static void displaySensorData(int soil, int rain, int light, float temp, float hum);
static void controlPump(int soilValue, int rainValue);
static void turnPumpOn();
static void turnPumpOff();
static void connectToWiFi();
static float convertSoilMoistureToPercentage(int rawValue);
static float convertLightToPercentage(int rawValue);
static bool sendSensorDataToBackend(int soilRaw, int rainRaw, int lightRaw, float temp, float hum);
static void printWifiDiag();

void setup() {
  Serial.begin(115200);
  delay(2000);

  Serial.println("\n====================================");
  Serial.println(" ESP32 Smart Irrigation + API");
  Serial.println("====================================\n");

  dht.begin();
  Serial.println("DHT initialized");

  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, LOW);
  Serial.println("Relay OFF");

  Serial.println("Analog inputs ready");

  Serial.println("\nConnecting to WiFi...");
  connectToWiFi();

  if (wifiConnected) {
    Serial.println("WiFi OK");
    printWifiDiag();
    Serial.print("RSSI: ");
    Serial.print(WiFi.RSSI());
    Serial.println(" dBm");
    Serial.print("device_id: ");
    Serial.println(deviceId);
    Serial.print("POST target: ");
    Serial.print(BACKEND_USE_HTTPS ? "https://" : "http://");
    Serial.print(BACKEND_HOST);
    Serial.println(API_PATH);
  } else {
    Serial.println("WiFi failed - local pump logic only");
  }

  Serial.println("\nStarting loop.\n");
  delay(1000);
}

void loop() {
  unsigned long now = millis();

  if (WiFi.status() == WL_CONNECTED) {
    wifiConnected = true;
    lastWifiAttemptMs = now;
  } else {
    wifiConnected = false;
    if (now - lastWifiAttemptMs >= WIFI_RETRY_INTERVAL_MS) {
      Serial.println("WiFi down - reconnecting...");
      connectToWiFi();
      lastWifiAttemptMs = millis();
    }
  }

  bool needRead = (now - lastReadTime >= READING_INTERVAL);
  bool enoughTimeSinceSuccess = (now - lastBackendSendTime >= BACKEND_SEND_INTERVAL);
  bool enoughTimeSinceAttempt = (now - lastBackendAttemptTime >= BACKEND_RETRY_INTERVAL);
  bool needSend = wifiConnected && enoughTimeSinceSuccess && enoughTimeSinceAttempt;

  if (!needRead && !needSend) {
    delay(50);
    return;
  }

  int soilValue, rainValue, lightValue;
  float temperature, humidity;

  if (needRead) {
    lastReadTime = now;
  }
  readAllSensors(soilValue, rainValue, lightValue, temperature, humidity);

  if (needRead) {
    displaySensorData(soilValue, rainValue, lightValue, temperature, humidity);
    controlPump(soilValue, rainValue);
  }

  if (needSend) {
    lastBackendAttemptTime = now;
    bool ok = sendSensorDataToBackend(soilValue, rainValue, lightValue, temperature, humidity);
    if (ok) {
      lastBackendSendTime = now;
    }
    Serial.println(ok ? "Backend: OK\n" : "Backend: FAIL\n");
  }

  delay(50);
}

int readAnalogSmoothed(uint8_t pin) {
  long sum = 0;
  for (int i = 0; i < ANALOG_SAMPLES; i++) {
    sum += analogRead(pin);
    delayMicroseconds(150);
  }
  return (int)(sum / ANALOG_SAMPLES);
}

bool readDht(float& tempC, float& humPct) {
  humPct = dht.readHumidity();
  tempC = dht.readTemperature();
  if (isnan(humPct) || isnan(tempC)) {
    tempC = -999;
    humPct = -999;
    return false;
  }
  return true;
}

void readAllSensors(int& soil, int& rain, int& light, float& tempC, float& humPct) {
  soil = readAnalogSmoothed(SOIL_PIN);
  rain = readAnalogSmoothed(RAIN_PIN);
  light = readAnalogSmoothed(LIGHT_PIN);
  readDht(tempC, humPct);
}

void displaySensorData(int soil, int rain, int light, float temp, float hum) {
  Serial.println("-------------------------------------");
  Serial.print("Soil ADC: ");
  Serial.print(soil);
  Serial.print("  ");
  if (soil > SOIL_DRY_THRESHOLD) {
    Serial.println("(dry)");
  } else if (soil > 1500) {
    Serial.println("(moderate)");
  } else {
    Serial.println("(wet)");
  }

  Serial.print("Rain ADC: ");
  Serial.print(rain);
  Serial.println(rain < RAIN_THRESHOLD ? "  (rain/wet pad)" : "  (no rain)");

  Serial.print("Light ADC: ");
  Serial.print(light);
  Serial.print("  ");
  if (light > LIGHT_DARK_THRESHOLD) {
    Serial.println("(dark)");
  } else if (light > 1500) {
    Serial.println("(moderate)");
  } else {
    Serial.println("(bright)");
  }

  Serial.print("Temp: ");
  if (temp != -999) {
    Serial.print(temp, 1);
    Serial.println(" C");
  } else {
    Serial.println("ERROR");
  }

  Serial.print("Humidity: ");
  if (hum != -999) {
    Serial.print(hum, 1);
    Serial.println(" %");
  } else {
    Serial.println("ERROR");
  }

  Serial.print("Pump: ");
  Serial.println(pumpRunning ? "ON" : "OFF");
  Serial.println("-------------------------------------\n");
}

void controlPump(int soilValue, int rainValue) {
  bool soilIsDry = (soilValue > SOIL_DRY_THRESHOLD);
  bool notRaining = (rainValue > RAIN_THRESHOLD);

  if (soilIsDry && notRaining && !pumpRunning) {
    turnPumpOn();
    Serial.println("AUTO: Pump ON (dry soil, no rain)");
    return;
  }

  if (!pumpRunning) {
    return;
  }

  bool wantOff = (!soilIsDry || !notRaining);
  if (!wantOff) {
    return;
  }

  if (!notRaining) {
    turnPumpOff();
    Serial.println("AUTO: Pump OFF (rain)");
    return;
  }

  if (millis() - pumpStartTime >= PUMP_MIN_RUN_TIME) {
    turnPumpOff();
    Serial.println("AUTO: Pump OFF (soil wet enough)");
  }
}

void turnPumpOn() {
  digitalWrite(RELAY_PIN, HIGH);
  pumpRunning = true;
  pumpStartTime = millis();
}

void turnPumpOff() {
  digitalWrite(RELAY_PIN, LOW);
  pumpRunning = false;
}

void connectToWiFi() {
  if (WiFi.status() == WL_CONNECTED) {
    wifiConnected = true;
    return;
  }

  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);

  Serial.print("WiFi");
  unsigned long startTime = millis();
  while (WiFi.status() != WL_CONNECTED && (millis() - startTime < 12000)) {
    delay(400);
    Serial.print(".");
  }
  Serial.println();

  wifiConnected = (WiFi.status() == WL_CONNECTED);
  if (!wifiConnected) {
    Serial.println("WiFi connect failed");
  }
}

float convertSoilMoistureToPercentage(int rawValue) {
  float percentage = (float)map(rawValue, 0, 4095, 100, 0);
  return constrain(percentage, 0.0f, 100.0f);
}

float convertLightToPercentage(int rawValue) {
  float percentage = (float)map(rawValue, 0, 4095, 100, 0);
  return constrain(percentage, 0.0f, 100.0f);
}

void printWifiDiag() {
  if (WiFi.status() != WL_CONNECTED) {
    return;
  }
  Serial.print("ESP32 IP: ");
  Serial.println(WiFi.localIP());
  Serial.print("Gateway:  ");
  Serial.println(WiFi.gatewayIP());
  Serial.print("Mask:     ");
  Serial.println(WiFi.subnetMask());

  IPAddress resolved;
  if (WiFi.hostByName(BACKEND_HOST, resolved)) {
    Serial.print("Resolved backend IP: ");
    Serial.println(resolved);
  } else {
    Serial.println("DNS lookup failed for backend host.");
  }
}

bool sendSensorDataToBackend(int soilRaw, int rainRaw, int lightRaw, float temp, float hum) {
  if (WiFi.status() != WL_CONNECTED) {
    return false;
  }

  // Keep client objects alive for the full HTTP request lifecycle.
  WiFiClientSecure secureClient;
  WiFiClient plainClient;
  HTTPClient http;
  float soilMoisture = convertSoilMoistureToPercentage(soilRaw);
  float lightIntensity = convertLightToPercentage(lightRaw);
  bool raining = (rainRaw < RAIN_THRESHOLD);

  StaticJsonDocument<384> doc;
  doc["device_id"] = deviceId;
  doc["soil_moisture"] = soilMoisture;
  doc["light_intensity"] = lightIntensity;
  if (temp != -999) {
    doc["temperature"] = temp;
  }
  if (hum != -999) {
    doc["humidity"] = hum;
  }
  doc["rainfall"] = raining;
  doc["pump_on"] = pumpRunning ? 1 : 0;

  String jsonString;
  serializeJson(doc, jsonString);

  Serial.println("POST sensor reading...");
  Serial.println(jsonString);

  String endpoint = String(BACKEND_USE_HTTPS ? "https://" : "http://") + BACKEND_HOST + API_PATH;

  bool beginOk = false;
  if (BACKEND_USE_HTTPS) {
    secureClient.setInsecure();
    beginOk = http.begin(secureClient, endpoint);
  } else {
    beginOk = http.begin(plainClient, endpoint);
  }

  if (!beginOk) {
    Serial.println("HTTP begin() failed");
    return false;
  }

  http.setFollowRedirects(HTTPC_STRICT_FOLLOW_REDIRECTS);
  const char* headerKeys[] = {"Location"};
  http.collectHeaders(headerKeys, 1);
  http.addHeader("Content-Type", "application/json");
  http.setConnectTimeout(8000);
  http.setTimeout(10000);

  int code = http.POST(jsonString);
  bool success = false;

  if (code > 0) {
    Serial.printf("HTTP %d\n", code);
    if (code == 301 || code == 302 || code == 307 || code == 308) {
      Serial.print("Redirect to: ");
      Serial.println(http.header("Location"));
    }
    String response = http.getString();
    if (response.length()) {
      Serial.println(response);
    }
    if (code == 200 || code == 201) {
      success = true;
    }
  } else {
    Serial.printf("HTTP error %d: %s\n", code, http.errorToString(code).c_str());
  }

  http.end();
  return success;
}
